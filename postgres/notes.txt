drop database (dbname); - to delete particular database

CRUD operation - create, read, update, delete

// creating table - 
create table (table_name)(
    id int,
    name varchar(100),
    city varchar(100)
);
// adding data into table - 
insert into (table_name)(id, name, city) values (101, 'xyz', 'Delhi');
// reading data from table -
select * from (table_name);
// updating the data -
update (table_name) set city='Bangalore' where id=2;
// deleting data - 
delete from (table_name) where name='xyz';
